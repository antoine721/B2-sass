structure:

projet/

├── sass/
│   ├── main.scss           # fichier principal Sass
│   ├── _variables.scss     # variables (ex : couleurs, espacements)
│   ├── _mixins.scss        # mixins
│   ├── _base.scss          # styles de base (reset, typographie)
│   └── _components/
│       ├── _header.scss    # exemple de fichier de composant
│       └── _footer.scss    # autre composant
└── index.html
